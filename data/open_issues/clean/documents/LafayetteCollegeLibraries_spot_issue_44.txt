add multilingual indexing
We have data/fields in multiple languages:  English Chinese Japanese Korean French German  These will need to have tokenizing setup in solr/config/schema.xml. For example, to enable a Japanese text field,  add the following to <types>:  <fieldType name="text_ja" class="solr.TextField" positionIncrementGap="100" autoGeneratePhraseQueries="false">   <analyzer>     <!-- Uncomment if you need to handle iteration marks: -->     <!-- <charFilter class="solr.JapaneseIterationMarkCharFilterFactory" /> -->     <tokenizer class="solr.JapaneseTokenizerFactory" mode="search" userDictionary="lang/userdict_ja.txt"/>     <filter class="solr.JapaneseBaseFormFilterFactory"/>     <filter class="solr.JapanesePartOfSpeechStopFilterFactory" tags="lang/stoptags_ja.txt"/>     <filter class="solr.CJKWidthFilterFactory"/>     <filter class="solr.StopFilterFactory" ignoreCase="true" words="lang/stopwords_ja.txt"/>     <filter class="solr.JapaneseKatakanaStemFilterFactory" minimumLength="4"/>     <filter class="solr.LowerCaseFilterFactory"/>   </analyzer> </fieldType>  add the following to <fields>:  <!-- Japanese text (tj...) --> <dynamicField name="*_tji" type="text_ja" stored="false" indexed="true" multiValued="false"/> <dynamicField name="*_tjim" type="text_ja" stored="false" indexed="true" multiValued="true"/> <dynamicField name="*_tjs" type="text_ja" stored="true" indexed="false" multiValued="false"/> <dynamicField name="*_tjsm" type="text_ja" stored="true" indexed="false" multiValued="true"/> <dynamicField name="*_tjsi" type="text_ja" stored="true" indexed="true" multiValued="false"/> <dynamicField name="*_tjsim" type="text_ja" stored="true" indexed="true" multiValued="true"/> <dynamicField name="*_tjiv" type="text_ja" stored="false" indexed="true" multiValued="false" termVectors="true" termPositions="true" termOffsets="true"/> <dynamicField name="*_tjimv" type="text_ja" stored="false" indexed="true" multiValued="true" termVectors="true" termPositions="true" termOffsets="true"/> <dynamicField name="*_tjsiv" type="text_ja" stored="true" indexed="true" multiValued="false" termVectors="true" termPositions="true" termOffsets="true"/> <dynamicField name="*_tjsimv" type="text_ja" stored="true" indexed="true" multiValued="true" termVectors="true" termPositions="true" termOffsets="true"/>
